# Phase 3.3 Messages Panel Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a Messages panel with Combat, Espionage, and Fleet tabs backed by existing `combatLog`/`espionageReports` plus a new `FleetNotification` type for harvest/transport completions.

**Architecture:** New `FleetNotification` model written into `state.fleetNotifications[]` at mission arrival time (not completion). `EspionageReport` gains a `detectionChance` field. Context API is replaced (not extended) with symmetric per-tab read/delete actions. `MessagesPanel.tsx` is a pure React component reading from context.

**Tech Stack:** TypeScript (strict), React 19, vitest, @testing-library/react, jsdom

Add a helper function near the top of the file (after imports):

```ts
function pushFleetNotification(
  state: GameState,
  mission: FleetMission,
  targetName: string,
  loot: { metal: number; crystal: number; deuterium: number },
): void {
  const alreadyNotified = state.fleetNotifications.some((n) => n.missionId === mission.id);
  if (alreadyNotified) return;

  state.fleetNotifications.push({
    id: `${mission.id}-notif`,
    missionId: mission.id,
    timestamp: Date.now(),
    missionType: mission.type as 'harvest' | 'transport',
    targetCoordinates: { ...mission.targetCoordinates },
    targetName,
    loot: { ...loot },
    read: false,
  });
}
```

import { useState } from 'react';
import type { CombatLogEntry, EspionageReport, FleetNotification } from '../models/Fleet.ts';
import { SHIPS } from '../data/ships.ts';
import { DEFENCES } from '../data/defences.ts';
import { useGame } from '../context/GameContext';
import { formatNumber } from '../utils/format.ts';

type Tab = 'combat' | 'espionage' | 'fleet';

function formatCoords(c: { galaxy: number; system: number; slot: number }): string {
  return `G:${c.galaxy} S:${c.system} P:${c.slot}`;
}

function formatDate(ts: number): string {
  return new Date(ts).toLocaleString([], {
    month: 'short', day: 'numeric',
    hour: '2-digit', minute: '2-digit',
  });
}

function formatShipLosses(snapshot: { ships?: Partial<Record<string, number>>; defences?: Partial<Record<string, number>> }): string[] {
  const lines: string[] = [];
  for (const [id, count] of Object.entries(snapshot.ships ?? {})) {
    if ((count ?? 0) <= 0) continue;
    lines.push(`${count}× ${SHIPS[id as keyof typeof SHIPS]?.name ?? id}`);
  }
  for (const [id, count] of Object.entries(snapshot.defences ?? {})) {
    if ((count ?? 0) <= 0) continue;
    lines.push(`${count}× ${DEFENCES[id as keyof typeof DEFENCES]?.name ?? id}`);
  }
  return lines;
}

// ── Combat Tab ────────────────────────────────────────────────────────────────

function CombatRow({ entry, onRead, onDelete }: {
  entry: CombatLogEntry;
  onRead: (id: string) => void;
  onDelete: (id: string) => void;
}) {
  const [expanded, setExpanded] = useState(false);
  const { result } = entry;
  const outcomeLabel =
    result.outcome === 'attacker_wins' ? 'Victory' :
    result.outcome === 'defender_wins' ? 'Defeat' : 'Draw';

  const toggle = () => {
    setExpanded((prev) => !prev);
    if (!entry.read) onRead(entry.id);
  };

  const attackerLossLines = formatShipLosses(result.attackerLosses);
  const defenderLossLines = formatShipLosses(result.defenderLosses);
  const rebuiltLines = Object.entries(result.defencesRebuilt)
    .filter(([, count]) => (count ?? 0) > 0)
    .map(([id, count]) => `${count}× ${DEFENCES[id as keyof typeof DEFENCES]?.name ?? id}`);

  return (
    <div className={`message-row ${entry.read ? '' : 'message-unread'}`}>
      <div className="message-row-header" onClick={toggle} role="button" tabIndex={0} onKeyDown={(e) => e.key === 'Enter' && toggle()}>
        <span className="message-row-icon">⚔</span>
        <span className="message-row-summary">
          [{formatCoords(entry.targetCoordinates)}] — {outcomeLabel} — {result.rounds} round{result.rounds !== 1 ? 's' : ''}
          {' '}— Loot: M {formatNumber(result.loot.metal)} | C {formatNumber(result.loot.crystal)} | D {formatNumber(result.loot.deuterium)}
        </span>
        <span className="message-row-date">{formatDate(entry.timestamp)}</span>
        {!entry.read && <span className="message-unread-dot" aria-label="unread">●</span>}
        <button type="button" className="message-delete-btn" aria-label="delete" onClick={(e) => { e.stopPropagation(); onDelete(entry.id); }}>×</button>
      </div>
      {expanded && (
        <div className="message-detail">
          <p><strong>Outcome:</strong> {outcomeLabel} ({result.rounds} rounds)</p>
          {attackerLossLines.length > 0
            ? <p><strong>Attacker losses:</strong> {attackerLossLines.join(', ')}</p>
            : <p><strong>Attacker losses:</strong> None</p>}
          {defenderLossLines.length > 0
            ? <p><strong>Defender losses:</strong> {defenderLossLines.join(', ')}</p>
            : <p><strong>Defender losses:</strong> None</p>}
          {rebuiltLines.length > 0 && <p><strong>Rebuilt:</strong> {rebuiltLines.join(', ')}</p>}
          <p><strong>Loot:</strong> M {formatNumber(result.loot.metal)} | C {formatNumber(result.loot.crystal)} | D {formatNumber(result.loot.deuterium)}</p>
          {(result.debrisCreated.metal > 0 || result.debrisCreated.crystal > 0) && (
            <p><strong>Debris created:</strong> M {formatNumber(result.debrisCreated.metal)} | C {formatNumber(result.debrisCreated.crystal)}</p>
          )}
        </div>
      )}
    </div>
  );
}

// ── Espionage Tab ─────────────────────────────────────────────────────────────

function EspionageRow({ report, onRead, onDelete }: {
  report: EspionageReport;
  onRead: (id: string) => void;
  onDelete: (id: string) => void;
}) {
  const [expanded, setExpanded] = useState(false);

  const toggle = () => {
    setExpanded((prev) => !prev);
    if (!report.read) onRead(report.id);
  };

  const resourceLine = report.resources
    ? `M ${formatNumber(report.resources.metal)} | C ${formatNumber(report.resources.crystal)} | D ${formatNumber(report.resources.deuterium)}`
    : 'Unknown';

  return (
    <div className={`message-row ${report.read ? '' : 'message-unread'}`}>
      <div className="message-row-header" onClick={toggle} role="button" tabIndex={0} onKeyDown={(e) => e.key === 'Enter' && toggle()}>
        <span className="message-row-icon">🔭</span>
        <span className="message-row-summary">
          {report.targetName} [{formatCoords(report.targetCoordinates)}]
          {report.detected ? ' — Detected!' : ` — Resources: ${resourceLine}`}
        </span>
        <span className="message-row-date">{formatDate(report.timestamp)}</span>
        {!report.read && <span className="message-unread-dot" aria-label="unread">●</span>}
        <button type="button" className="message-delete-btn" aria-label="delete" onClick={(e) => { e.stopPropagation(); onDelete(report.id); }}>×</button>
      </div>
      {expanded && (
        <div className="message-detail">
          {report.detected && <p className="hint danger">Probes detected and destroyed ({report.probesLost}/{report.probesSent})</p>}
          <p><strong>Counter-espionage chance:</strong> {(report.detectionChance * 100).toFixed(1)}%</p>
          {report.resources
            ? <p><strong>Resources:</strong> M {formatNumber(report.resources.metal)} | C {formatNumber(report.resources.crystal)} | D {formatNumber(report.resources.deuterium)}</p>
            : <p><strong>Resources:</strong> Unknown (insufficient tech level)</p>}
          {report.fleet !== undefined
            ? <p><strong>Fleet:</strong> {Object.entries(report.fleet).filter(([,c]) => (c??0)>0).map(([id,c]) => `${c}× ${SHIPS[id as keyof typeof SHIPS]?.name ?? id}`).join(', ') || 'None'}</p>
            : <p><strong>Fleet:</strong> Unknown (insufficient tech level)</p>}
          {report.defences !== undefined
            ? <p><strong>Defences:</strong> {Object.entries(report.defences).filter(([,c]) => (c??0)>0).map(([id,c]) => `${c}× ${DEFENCES[id as keyof typeof DEFENCES]?.name ?? id}`).join(', ') || 'None'}</p>
            : <p><strong>Defences:</strong> Unknown (insufficient tech level)</p>}
          {report.tier !== undefined && <p><strong>Tier:</strong> {report.tier}{report.specialty ? ` (${report.specialty})` : ''}</p>}
          {report.rebuildStatus && (
            <p><strong>Rebuild status:</strong> Fleet {report.rebuildStatus.fleetPct}% | Defence {report.rebuildStatus.defencePct}%</p>
          )}
        </div>
      )}
    </div>
  );
}

// ── Fleet Tab ─────────────────────────────────────────────────────────────────

function FleetRow({ notif, onRead, onDelete }: {
  notif: FleetNotification;
  onRead: (id: string) => void;
  onDelete: (id: string) => void;
}) {
  const [expanded, setExpanded] = useState(false);
  const hasLoot = notif.loot.metal > 0 || notif.loot.crystal > 0 || notif.loot.deuterium > 0;

  const toggle = () => {
    setExpanded((prev) => !prev);
    if (!notif.read) onRead(notif.id);
  };

  const summaryLabel = notif.missionType === 'transport'
    ? `Transport delivered to ${notif.targetName}`
    : `Harvest returned from ${notif.targetName}`;

  return (
    <div className={`message-row ${notif.read ? '' : 'message-unread'}`}>
      <div className="message-row-header" onClick={toggle} role="button" tabIndex={0} onKeyDown={(e) => e.key === 'Enter' && toggle()}>
        <span className="message-row-icon">🚀</span>
        <span className="message-row-summary">{summaryLabel}</span>
        <span className="message-row-date">{formatDate(notif.timestamp)}</span>
        {!notif.read && <span className="message-unread-dot" aria-label="unread">●</span>}
        <button type="button" className="message-delete-btn" aria-label="delete" onClick={(e) => { e.stopPropagation(); onDelete(notif.id); }}>×</button>
      </div>
      {expanded && (
        <div className="message-detail">
          <p><strong>Target:</strong> {notif.targetName} [{formatCoords(notif.targetCoordinates)}]</p>
          {hasLoot
            ? <p><strong>Resources:</strong> M {formatNumber(notif.loot.metal)} | C {formatNumber(notif.loot.crystal)} | D {formatNumber(notif.loot.deuterium)}</p>
            : <p className="hint">{notif.missionType === 'harvest' ? 'Debris field was already cleared.' : 'Destination planet no longer exists.'}</p>}
        </div>
      )}
    </div>
  );
}

// ── Panel ─────────────────────────────────────────────────────────────────────

export function MessagesPanel() {
  const {
    gameState,
    fleetNotifications,
    markCombatRead, markAllCombatRead, deleteCombatEntry,
    markEspionageRead, markAllEspionageRead, deleteEspionageReport,
    markFleetRead, markAllFleetRead, deleteFleetNotification,
  } = useGame();

  const [activeTab, setActiveTab] = useState<Tab>('combat');

  const sortedCombat = [...gameState.combatLog].sort((a, b) => b.timestamp - a.timestamp);
  const sortedEspionage = [...gameState.espionageReports].sort((a, b) => b.timestamp - a.timestamp);
  const sortedFleet = [...fleetNotifications].sort((a, b) => b.timestamp - a.timestamp);

  const combatUnread = gameState.combatLog.filter((e) => !e.read).length;
  const espionageUnread = gameState.espionageReports.filter((r) => !r.read).length;
  const fleetUnread = fleetNotifications.filter((n) => !n.read).length;

  return (
    <section className="panel">
      <h1 className="panel-title">Messages</h1>

      <div className="messages-tabs">
        {(['combat', 'espionage', 'fleet'] as const).map((tab) => {
          const unread = tab === 'combat' ? combatUnread : tab === 'espionage' ? espionageUnread : fleetUnread;
          return (
            <button
              key={tab}
              type="button"
              className={`messages-tab-btn ${activeTab === tab ? 'active' : ''}`}
              onClick={() => setActiveTab(tab)}
            >
              {tab.charAt(0).toUpperCase() + tab.slice(1)}
              {unread > 0 && <span className="nav-badge">{unread}</span>}
            </button>
          );
        })}
      </div>

      <div className="panel-card">
        {activeTab === 'combat' && (
          <>
            <div className="messages-tab-header">
              <h2 className="section-title">Combat Reports</h2>
              <button type="button" className="btn btn-sm" onClick={markAllCombatRead}>Mark all read</button>
            </div>
            {sortedCombat.length === 0
              ? <p className="hint">No combat messages</p>
              : sortedCombat.map((entry) => (
                  <CombatRow key={entry.id} entry={entry} onRead={markCombatRead} onDelete={deleteCombatEntry} />
                ))}
          </>
        )}

        {activeTab === 'espionage' && (
          <>
            <div className="messages-tab-header">
              <h2 className="section-title">Espionage Reports</h2>
              <button type="button" className="btn btn-sm" onClick={markAllEspionageRead}>Mark all read</button>
            </div>
            {sortedEspionage.length === 0
              ? <p className="hint">No espionage messages</p>
              : sortedEspionage.map((report) => (
                  <EspionageRow key={report.id} report={report} onRead={markEspionageRead} onDelete={deleteEspionageReport} />
                ))}
          </>
        )}

        {activeTab === 'fleet' && (
          <>
            <div className="messages-tab-header">
              <h2 className="section-title">Fleet Notifications</h2>
              <button type="button" className="btn btn-sm" onClick={markAllFleetRead}>Mark all read</button>
            </div>
            {sortedFleet.length === 0
              ? <p className="hint">No fleet messages</p>
              : sortedFleet.map((notif) => (
                  <FleetRow key={notif.id} notif={notif} onRead={markFleetRead} onDelete={deleteFleetNotification} />
                ))}
          </>
        )}
      </div>
    </section>
  );
}
