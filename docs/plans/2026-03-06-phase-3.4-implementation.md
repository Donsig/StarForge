// Replace the existing private function:
// function researchLabLevelForItem(state: GameState, item: QueueItem): number { ... }
// With:

export function effectiveResearchLabLevel(state: GameState, item: QueueItem): number {
  const irnLevel = state.research.intergalacticResearchNetwork ?? 0;
  if (irnLevel === 0) {
    const src =
      state.planets[item.sourcePlanetIndex ?? state.activePlanetIndex] ??
      state.planets[state.activePlanetIndex] ??
      state.planets[0];
    return src?.buildings.researchLab ?? 0;
  }
  const topN = irnLevel + 1;
  return state.planets
    .map((p) => p.buildings.researchLab)
    .sort((a, b) => b - a)
    .slice(0, topN)
    .reduce((sum, lvl) => sum + lvl, 0);
}
```

import { useState, useEffect, useRef } from 'react';

export function OverviewPanel() {
  const { gameState, productionRates, renamePlanet } = useGame();
  const planet = gameState.planets[gameState.activePlanetIndex];
  const activePlanetIndex = gameState.activePlanetIndex;

  const [editing, setEditing] = useState(false);
  const [draft, setDraft] = useState('');
  const committedRef = useRef(false);

  // Exit edit mode when the active planet changes
  useEffect(() => {
    setEditing(false);
    setDraft('');
    committedRef.current = false;
  }, [activePlanetIndex]);

  const startEditing = () => {
    committedRef.current = false;
    setDraft(planet.name);
    setEditing(true);
  };

  const commit = () => {
    if (committedRef.current) return;
    committedRef.current = true;
    renamePlanet(activePlanetIndex, draft);
    setEditing(false);
  };

  const cancel = () => {
    setEditing(false);
    setDraft('');
    committedRef.current = false;
  };

  // ... rest of existing component

  // Replace the Name stat line:
  // Old: <span>{planet.name}</span>
  // New:
  return (
    <section className="panel">
      {/* ... existing JSX ... */}
      <p className="stat-line">
        <span className="label">Name</span>
        {editing ? (
          <input
            type="text"
            className="input planet-rename-input"
            value={draft}
            maxLength={30}
            autoFocus
            onChange={(e) => setDraft(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === 'Enter') { e.preventDefault(); commit(); }
              if (e.key === 'Escape') { e.preventDefault(); cancel(); }
            }}
            onBlur={commit}
            aria-label="Planet name"
          />
        ) : (
          <>
            <span>{planet.name}</span>
            <button
              type="button"
              className="btn btn-sm planet-rename-btn"
              aria-label="Rename planet"
              onClick={startEditing}
            >
              ✏
            </button>
          </>
        )}
      </p>
      {/* ... rest of JSX ... */}
    </section>
  );
}
```
